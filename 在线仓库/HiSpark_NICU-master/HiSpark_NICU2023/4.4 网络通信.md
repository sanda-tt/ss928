  ```c
  AT+RST
  AT+STARTSTA
  AT+SCAN
  AT+SCANRESULT
  AT+CONN="WiFi名称",,WiFi认证方式,"WiFi密码"
  AT+DHCP=wlan0,1
  AT+IFCFG
  AT+PING=目的IP地址
  AT+DISCONN
  ```
