模型可以通过以下代码完成快速推理
```
#include "model.h"
#include "log.h"

using namespace Infer;

int main()
{
    EnvInit();
    std::string omModelPath = "/path/to/model.om"; // 模型文件路径 
    std::string imagePath = "/path/to/file_list.json"; // 输入文本文件路径
    std::unique_ptr<Model> model = std::make_unique<Model>();
    if (model->Load(omModelPath, ModelType::FastSpeech2) != 0) {
        LOG(ERROR) << "fail to load model";
        return -1;
    }
    auto ret = model->Infer(imagePath, FileType::JsonFile);
    if (ret.size() == 0) {
        LOG(ERROR) << "fail to infer model";
        model->Unload();
        return -1;
    }
    if (model->Unload() != 0) {
        LOG(ERROR) << "fail to unload model";
        return -1;
    }
    EnvDeinit();
    return 0;
}

```
备注：上述C++代码依赖的动态库与头文件位于[/samples/common](https://gitee.com/HiSpark/modelzoo/tree/master/samples/common)目录下，编译相关配置参考[CMakeLists.txt](https://gitee.com/HiSpark/modelzoo/blob/master/samples/built-in/audio/FastSpeech2/src/CMakeLists.txt)。
