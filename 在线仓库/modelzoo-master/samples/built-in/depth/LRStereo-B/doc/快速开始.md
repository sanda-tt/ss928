模型可以通过以下代码完成快速推理
```
#include "model.h"
#include "log.h"

using namespace Infer;

int main()
{
    EnvInit();
    std::string omModelPath = "/path/to/model.om"; // 模型文件路径 
    std::string imagePath = "/path/to/image.jpg"; // 输入图片路径
    std::string omModelDisPath = "/path/to/model.om"; // 视差转距离模型文件路径 
    bool type = std::stoi(cfg["type"]) == 1; // 是否使用视差转距离模型
    std::unique_ptr<Model> modelStereo = std::make_unique<Model>();
    modelStereo->Load(omModelPath, LRStereo)
    std::unique_ptr<Model> modelDis = std::make_unique<Model>();
    if (type && modelDis->Load(dis_path, LRStereoDis) != 0) {
        LOG(ERROR) << "fail to load Dis model";
        return 0;
    }
    std::vector<std::vector<std::string>> fileLists = ParseFileList(imagePath);
    std::vector<std::vector<Tensor>> ret;
    std::vector<Tensor> result;
    for (size_t i = 0; i < fileLists.size(); ++i) {
        std::string inputString = BuildInputString(fileLists[i]);
        ret = modelStereo->Infer(inputString, FileType::SingelImageFile);
        if (type) {
            result = modelDis->Infer(ret[0], inputString);
        }
    }
    ret.clear();
    ret.shrink_to_fit();
    if (modelStereo->Unload() != 0) {
        LOG(ERROR) << "fail to unload model";
        EnvDeinit();
        return 0;
    }
    if (type) {
        result.clear();
        result.shrink_to_fit();
        if (modelDis->Unload() != 0) {
            LOG(ERROR) << "fail to unload model";
            EnvDeinit();
            return 0;
        }
    }
    EnvDeinit();
    return 0;
}
```
备注：上述C++代码依赖的动态库与头文件位于[/samples/common](https://gitee.com/HiSpark/modelzoo/tree/master/samples/common)目录下，编译相关配置参考[CMakeLists.txt](https://gitee.com/HiSpark/modelzoo/blob/master/samples/built-in/depth/LiteAndRobustStereo/src/CMakeLists.txt)。
