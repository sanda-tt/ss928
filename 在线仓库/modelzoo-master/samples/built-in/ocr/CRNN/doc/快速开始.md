CRNN模型可以通过以下代码完成快速推理
```
#include "model.h"
#include "log.h"

using namespace Infer;

int main()
{
    EnvInit();
    std::string omModelPath = "/path/to/model.om"; // CRNN模型文件路径 
    std::string imagePath = "/path/to/image.jpg"; // 输入图片路径
    std::unique_ptr<Model> model = std::make_unique<Model>();
    if (model->Load(omModelPath, ModelType::CRNN) != 0) {
        LOG(ERROR) << "fail to load model";
        return -1;
    }
    auto ret = model->Infer(imagePath, FileType::SingelImageFile);
    if (ret.size() == 0) {
        LOG(ERROR) << "fail to infer model";
        model->Unload();
        return -1;
    }
    if (model->Unload() != 0) {
        LOG(ERROR) << "fail to unload model";
        return -1;
    }
    EnvDeinit();
    return 0;
}
```
备注：头文件和动态库位于[/samples/common](https://gitee.com/HiSpark/modelzoo/tree/master/samples/common)目录下，编译配置参考[文件](https://gitee.com/HiSpark/modelzoo/blob/master/samples/built-in/ocr/CRNN/src/CMakeLists.txt)
